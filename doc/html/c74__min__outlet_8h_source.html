16  // outlet_do_send() wraps all Max API function calls for all types.
17  //
18  // Code that uses outlets send their data out of the outlet using a send() method.
19  // There are various factors that determine how this call gets translated into a call to Max's legacy API:
20  //
21  // * The type of data being sent (int, float, symbol, list, etc.)
22  // * The thread specification which indicates which thread(s) are valid targets for the outlet call.
23  // * The thread action which indicates if/how to check the thread and what to do if it doesn't match the specification.
24  //
25  // Regardless of how the threading is handled, when it comes time to actually send the data to the outlet the
26  // outlet_do_send() helper function is called.
27 
28  template<typename outlet_type>
29  inline void outlet_do_send(t_max_outlet maxoutlet, const outlet_type& value) {
30  if (value[0].a_type == max::A_LONG || value[0].a_type == max::A_FLOAT)
31  max::outlet_list(maxoutlet, nullptr, static_cast<short>(value.size()), static_cast<const max::t_atom*>(&value[0]));
32  else {
33  if (value.size() > 1)
34  max::outlet_anything(maxoutlet, value[0], static_cast<short>(value.size() - 1), static_cast<const max::t_atom*>(&value[1]));
35  else
36  max::outlet_anything(maxoutlet, value[0], 0, nullptr);
37  }
38  }

56  // Calls to send() on an outlet may deliver the data to the Max outlet in one of several ways.
57  // If the call is made on the thread that is specified in the outlet's Thread Specification
58  // then the call is executed immediately and synchronously.
59  // If the call is made on a different thread than what is specified in the outlet's Thread Specification
60  // then this outlet_queue class is invoked to handle the Thread Action.
61  //
62  // The possible actions to take when the thread is different than the specification are implemented below:
63  //
64  // * assert: The default action, crash when this condition occurs.
65  // * first: Schedule or defer the call to the appropriate thread,
66  // if multiple sends are received before the call is serviced then only send the first one.
67  // * last: Schedule or defer the call to the appropriate thread,
68  // if multiple sends are received before the call is serviced then only send the last one.
69  // * fifo: Schedule or defer the call to the appropriate thread,
70  // if multiple sends are received before the call is serviced they are put in a fifo and all are delivered.
71  //
72  // The outlet_queue inherits from thread_trigger.
73  // The thread_trigger manages the internal t_qelem or t_clock used to trigger the callback.
74
