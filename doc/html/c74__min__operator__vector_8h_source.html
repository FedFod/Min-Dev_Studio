262  // SFINAE implementation used internally to determine if the Min class has a member named dspsetup.
263  // NOTE: This relies on the C++ member name being "dspsetup" -- not just the Max message name being "dspsetup".
264  // See the min.buffer.loop~ object for an example.
265  //
266  // To test this in isolation, for a class named slide, use the following code:
267  // static_assert(has_dspsetup<slide>::value, "error");

269  template<typename min_class_type>
270  struct has_dspsetup {
271  template<class, class>
272  class checker;
273 
274  template<typename C>
275  static std::true_type test(checker<C, decltype(&C::dspsetup)>*);
276 
277  template<typename C>
278  static std::false_type test(...);
279 
280  typedef decltype(test<min_class_type>(nullptr)) type;

290  // The min_dsp64_io function handles updating the inlet and outlet connection state any time the dsp64 message is called.
291 
292  template<class min_class_type>
293  void min_dsp64_io(minwrap<min_class_type>* self, short* count) {
294  int i = 0;
295 
296  while (i < self->m_min_object.inlets().size()) {
297  self->m_min_object.inlets()[i]->update_signal_connection(count[i] != 0);
298  ++i;
299  }
300  while (i < self->m_min_object.outlets().size()) {
301  self->m_min_object.outlets()[i - self->m_min_object.inlets().size()]->update_signal_connection(count[i] != 0);
302  ++i;
303  }
304  }
