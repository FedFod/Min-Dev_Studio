114  inline std::string doc_format(const std::string& input) {
115  strings tokens = str::split(input, ' ');
116 
117  for (auto& token : tokens) {
118  if (token[0] == '@') { // attribute
119  token = string("<at>") + token.substr(1, string::npos) + string("</at>");
120  }
121  else if (token[0] == '#') { // message
122  token = string("<m>") + token.substr(1, string::npos) + string("</m>");
123  }
124  else if (token[0] == '[' && token[token.size() - 1] == ']') { // object
125  token = string("<o>") + token.substr(1, token.size() - 2) + string("</o>");
126  }
127  else if (token == "&") {
128  token = string("&amp;");
129  }
130  }
131 
132  return str::join(tokens);
133  }
