414  std::string smaxname;
415 
416  const char* start = strrchr(maxname, '/'); // mac paths
417  if (start)

428  if (end) {

431  }
432  else { // all other objects
433  const char* end = strrchr(start, '.');
434  if (!end)
435  end = start + strlen(start);
